import {animOut, animIn, processClickData} from "../modules/transitions";

/**
 * a component that allows you to put steps on a page and animate navigation
 * @module
 *
 */
export default {
	name: "basic_stepper",
	/**
	 * @return {Object} the data
	 * @return {Number} step the current step
	 * @property {Number} step the current step
	 */
	data: function () {
		/** @lends basicStepperComponent#data */ return {
			step: 0
		}
	},
	/**
	 * @type {Object}
	 */
	props: {
		/**
		 * a class to add to steps
		 * @type {String}
		 */
		stepClass: String,
		/**
		 * whether step transitions should animate
		 * @type {Object}
		 * @default  true
		 */
		animate: {
			type: Boolean,
			default: true
		},
		/**
		 * default animation props to apply to animations
		 * @type {Object}
		 */
		defaultProps: {
			type: Object
		}
	},
	/**
	 * @type {Object}
	 * @property {function} setStep
	 */
	methods: {
		/**
		 * set what step the stepper is on (click handler for navigation)
		 * @memberOf basicStepperComponent#methods
		 * @param {number} newStep   the index of the step to go to
		 * @param {Object} [animProps= basicStepperComponent#defaultProps ] properties to pass to {@link module:transitions~prop_defaults}
		 */
		setStep(newStep, animProps) {
			if (!this.animate) {
				this.step = newStep;
				return;
			}
			let props = Object.assign({}, this.defaultProps, animProps)
			processClickData(props)
			animOut(() => {
				this.step = newStep;
				this.$nextTick(animIn)
			});
		}
	}
}
